-- ===== ROOM FUNCTION =====
function editRoom(originalValue, newValue, label)
  gg.clearResults()
  gg.searchNumber(tostring(originalValue), gg.TYPE_QWORD, false)
  local results = gg.getResults(20)
  if #results > 0 then
    for i=1,#results do results[i].value = newValue end
    gg.setValues(results)
    gg.toast("Sala " .. label .. " editada a " .. newValue .. " ✅")
  else
    gg.toast("No se encontró la sala " .. label .. " ❌")
  end
  gg.clearResults()
end

-- ===== LEVEL MENU =====
function setLevel(newLevel)
  if newLevel < 1 or newLevel > 500 then
    gg.toast("El nivel debe estar entre 1 y 500 ⚠️")
    return
  end
  gg.clearResults()
  gg.searchNumber("1~500", gg.TYPE_DWORD, false)
  local results = gg.getResults(10)
  if #results > 0 then
    for i=1,#results do results[i].value = newLevel end
    gg.setValues(results)
    gg.toast("Nivel establecido en " .. newLevel .. " ⭐✅")
  else
    gg.toast("No se encontró el nivel ❌")
  end
  gg.clearResults()
end

function levelMenu()
  local c = gg.multiChoice({
    "Set Level 99 ⭐",
    "Set Level 500 ⭐ (máx)",
    "Elegir otro nivel (1–500) ✍️",
    "Back"
  }, nil, "Level Menu ⭐")
  if c == nil then return end
  if c[1] then setLevel(99) end
  if c[2] then setLevel(500) end
  if c[3] then
    local input = gg.prompt({"Ingresa el nivel (1–500):"}, {100}, {"number"})
    if input and input[1] then setLevel(tonumber(input[1])) end
  end
end

-- ===== VIP MENU =====
function setVIP(value)
  gg.clearResults()
  gg.searchNumber("0~1", gg.TYPE_DWORD, false)
  local results = gg.getResults(5)
  if #results > 0 then
    for i=1,#results do results[i].value = value end
    gg.setValues(results)
    if value == 1 then
      gg.toast("VIP Activado 👑✅")
    else
      gg.toast("VIP Desactivado ❌")
    end
  else
    gg.toast("No se encontró el valor de VIP ⚠️")
  end
  gg.clearResults()
end
